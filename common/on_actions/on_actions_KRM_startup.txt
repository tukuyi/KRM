on_actions = {
    on_startup = {
		effect = {
			QIE = {
				set_party_name = {
					ideology = national_populist
					name = QIE_YCP_1_party
					long_name = QIE_YCP_1_party_long
				}
				QIE_li_huang = {
					retire = yes
				}
				QIE_zeng_qi = {
					retire = yes
				}
				SZC = {
					SZC_zhao_yusong = {
						set_nationality = QIE
					}
				}
			}
			SHX = {
				SHX_chang_yansheng = {
					retire = yes
				}
			}
			HNN = {
				HNN_zuo_shunsheng = {
					retire = yes
				}
			}
		}
	}

	on_ruling_party_change = {
		effect = {
			if = {
				limit = {
					tag = QIE
					has_completed_focus = QIE_restoration
					SZC = {
						has_completed_focus = SZC_YCP_the_ninth_congress_of_YCP
					}
				}
				SZC = {
					country_event = {
						id = NKRsichuanPI.17
					}
				}
			}
		}
	}

	on_state_control_changed = {
		effect = {
			if = {
				limit = { tag = SZC }
				has_completed_focus = SZC_YCP_the_ninth_congress_of_YCP
				if = {
					limit = {
						tag = YCP
						FROM.FROM = { state = 608 } #Jingzhao
						has_country_flag = YCP_zhaoyusong_fired
					}
					country_event = { id = NKRsichuanPI.16 days = 3 }
				}
				else_if = {
					limit = {
						tag = YCP
						FROM.FROM = { state = 608 } #Jingzhao
					}
					country_event = { id = NKRsichuanPI.15 days = 3 }
					
				}
			}
		}
	}
}
