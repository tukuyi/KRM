defined_text = {
	name = GetROCCoalitionPPCost
	text = {
		trigger = {
			# 執政黨或聯合黨不顯示成本，顯示空
			meta_trigger = {
				text = {
					[tokenbase] = {
						NOT = {
							check_variable = { THIS.parliament_gui_status = 3 }
						}
					}
				}
				tokenbase = "[?roc_cur_token.GetTokenKey]"
			}
		}
		localization_key = ROC_coalition_no_cost
	}
	text = {
		trigger = {
			# 在野黨，計算並顯示成本
			meta_trigger = {
				text = {
					[tokenbase] = {
						check_variable = { THIS.parliament_gui_status = 3 }
						set_temp_variable = { temp_cost_seats = THIS.parliament_gui_seats }
					}
				}
				tokenbase = "[?roc_cur_token.GetTokenKey]"
			}
			# 公式：20 + floor(seats/10)*3
			set_temp_variable = { temp_pp_cost = temp_cost_seats }
			divide_temp_variable = { temp_pp_cost = 10 }
			multiply_temp_variable = { temp_pp_cost = 3 }
			add_to_temp_variable = { temp_pp_cost = 20 }
		}
		# localization_key = "[?temp_pp_cost]"
		localization_key = ROC_coalition_costpp
	}
}

defined_text = {
	name = GetROCCoalitionAllPPCost
	text = {
		trigger = {
			set_temp_variable = { temp_allpp_cost = 0 }
			set_temp_variable = { temp_pp_cost = 0 }
			all_of = {
				array = ROC_party_tokens
				value = roc_cur_token
				meta_trigger = {
					text = {
						[tokenbase] = {
							if = {
								limit  = {
									AND = {
										check_variable = { THIS.parliament_gui_constructalig = 2 }
										check_variable = { THIS.parliament_gui_status = 3 }
									}
								}
								set_temp_variable = { temp_cost_seats = var:THIS.parliament_gui_seats }
								log = "[?parliament_gui_seats]"
							}
						}
					}
					tokenbase = "[?roc_cur_token.GetTokenKey]"
				}
				# 公式：35 + floor(seats/10)*3
				if = {
					limit = {
						check_variable = {
							temp_cost_seats > 0
						}
					}
					set_temp_variable = { temp_pp_cost = temp_cost_seats }
					divide_temp_variable = { temp_pp_cost = 10 }
					multiply_temp_variable = { temp_pp_cost = 3 }
					add_to_temp_variable = { temp_pp_cost = 35 }
					add_to_temp_variable = { temp_allpp_cost = temp_pp_cost }
					set_temp_variable = { temp_pp_cost = 0 }
					set_temp_variable = { temp_cost_seats = 0 }
				}
			}
			log = "[?temp_allpp_cost]"
		}
		localization_key = "[?temp_allpp_cost]"
	}
}
