defined_text = {
	name = GetROCVoteCalculate
	text = {
		trigger = {
			# ==========================================
			# --- 1. 计算当前角度 Angle (0 到 360 度) ---
			# ==========================================
			set_temp_variable = { angle = party_idx }
			multiply_temp_variable = { angle = 360 }
			divide_temp_variable = { angle = var:ROC_parties_max }

			# ==========================================
			# --- 2. 算 Sine (这次把 Sine 交给 X 坐标) ---
			# ==========================================
			set_temp_variable = { y_angle = temp_var:angle }
			set_temp_variable = { is_neg_y = 0 }

			if = {
				limit = { check_variable = { y_angle > 179.999 } }
				subtract_from_temp_variable = { y_angle = 180 }
				set_temp_variable = { is_neg_y = 1 }
			}

			set_temp_variable = { term_y = 180 }
			subtract_from_temp_variable = { term_y = temp_var:y_angle }

			set_temp_variable = { T_y = temp_var:y_angle }
			multiply_temp_variable = { T_y = temp_var:term_y }

			set_temp_variable = { num_y = temp_var:T_y }
			multiply_temp_variable = { num_y = 60 } # 这里 60 是半径150对应的比例，如果想改半径，修改这里即可

			set_temp_variable = { T_y_div_10 = temp_var:T_y }
			divide_temp_variable = { T_y_div_10 = 10 }
			set_temp_variable = { den_y = 4050 }
			subtract_from_temp_variable = { den_y = temp_var:T_y_div_10 }

			# 赋值给 X！
			set_temp_variable = { party_x = temp_var:num_y }
			divide_temp_variable = { party_x = temp_var:den_y }
			if = {
				limit = { check_variable = { is_neg_y > 0 } }
				multiply_temp_variable = { party_x = -1 }
			}

			# ==========================================
			# --- 3. 算 Cosine (这次把 Cosine 交给 Y 坐标) ---
			# ==========================================
			set_temp_variable = { x_angle = temp_var:angle }
			add_to_temp_variable = { x_angle = 90 }

			if = {
				limit = { check_variable = { x_angle > 359.999 } }
				subtract_from_temp_variable = { x_angle = 360 }
			}

			set_temp_variable = { is_neg_x = 0 }
			if = {
				limit = { check_variable = { x_angle > 179.999 } }
				subtract_from_temp_variable = { x_angle = 180 }
				set_temp_variable = { is_neg_x = 1 }
			}

			set_temp_variable = { term_x = 180 }
			subtract_from_temp_variable = { term_x = temp_var:x_angle }

			set_temp_variable = { T_x = temp_var:x_angle }
			multiply_temp_variable = { T_x = temp_var:term_x }

			set_temp_variable = { num_x = temp_var:T_x }
			multiply_temp_variable = { num_x = 60 }

			set_temp_variable = { T_x_div_10 = temp_var:T_x }
			divide_temp_variable = { T_x_div_10 = 10 }
			set_temp_variable = { den_x = 4050 }
			subtract_from_temp_variable = { den_x = temp_var:T_x_div_10 }

			# 赋值给 Y！
			set_temp_variable = { party_y = temp_var:num_x }
			divide_temp_variable = { party_y = temp_var:den_x }
			if = {
				limit = { check_variable = { is_neg_x > 0 } }
				multiply_temp_variable = { party_y = -1 }
			}

			# 【关键点】：因为 12点钟 在 UI 里是上方 (负数)，我们需要把 Y 整体反转！
			multiply_temp_variable = { party_y = -1 }

			#增加基础xy
			add_to_temp_variable = { party_x = 200 }
			add_to_temp_variable = { party_y = 200 }

			# ==========================================
			# --- 4. 打印检查 ---
			# ==========================================
			log = "Party Index: [?party_idx] | Angle: [?angle] | X: [?party_x] | Y: [?party_y]"
		}
		localization_key = "empty_key"
	}
