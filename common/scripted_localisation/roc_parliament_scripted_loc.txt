defined_text = {
	name = GetROCParliamentCoalitionSeatsNumber
	text = {
		trigger = {
			set_temp_variable = { coalition_seats = 0 }
			all_of = {
				array = ROC_party_tokens
				value = roc_cur_token
				meta_trigger = {
					text = {
						[tokenbase] = {
							if = {
								limit = {
									OR = {
										check_variable = {
											THIS.parliament_gui_status = 1
										}
										check_variable = {
											THIS.parliament_gui_status = 2
										}
									}
								}
								add_to_temp_variable = {
									coalition_seats = THIS.parliament_gui_seats
								}
							}
						}
					}
					tokenbase = "[?roc_cur_token.GetTokenKey]"
				}
			}
		}
		localization_key = "[?coalition_seats]"
	}
}
