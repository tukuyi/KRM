CSP_recruit_character_effect = {#招募
	if = {
		limit = {
			NOT = {
				THIS = { has_character_flag = is_dead }
			}
		}
		if = {
			limit = { THIS = { is_army_leader = yes } }
			demote_to_general = yes
			remove_trait = { trait = recently_promoted }
		}
		set_nationality = CSP
	}
}
