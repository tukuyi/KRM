ROC_parliament_reorder_parties = {
    # 创建临时数组存储排序后的结果
    clear_array = temp_ROC_party_tokens_reordered
    
    # 第一步：找到执政党并添加到临时数组
    set_temp_variable = { ruling_party_found = 0 }
    for_each_loop = {
        array = ROC_party_tokens
        value = roc_cur_token
        meta_effect = {
            text = {
                [tokenbase] = {
					if = {
						limit = {
							check_variable = {
								THIS.parliament_gui_status = 1
							}
						}
						add_to_array = { temp_ROC_party_tokens_reordered = var:parliament_gui_party }
						set_temp_variable = { ruling_party_found = 1 }
					}
                }
            }
            tokenbase = "[?roc_cur_token.GetTokenKey]"
        }
    }
    
    # 第二步：添加所有联合党派
    for_each_loop = {
        array = ROC_party_tokens
        value = roc_cur_token
        meta_trigger = {
            text = {
                [tokenbase] = {
					if = {
						limit = {
							check_variable = {
								THIS.parliament_gui_status = 2
							}
						}
						add_to_array = { temp_ROC_party_tokens_reordered = var:parliament_gui_party }
					}
                }
            }
            tokenbase = "[?roc_cur_token.GetTokenKey]"
        }
    }
    
    # 第三步：添加所有在野党
    for_each_loop = {
        array = ROC_party_tokens
        value = roc_cur_token
        meta_trigger = {
            text = {
                [tokenbase] = {
					if = {
						limit = {
							check_variable = {
								THIS.parliament_gui_status = 3
							}
						}
						add_to_array = { temp_ROC_party_tokens_reordered = var:parliament_gui_party }
					}
                }
            }
            tokenbase = "[?roc_cur_token.GetTokenKey]"
        }
    }
    
    # 第四步：用排序后的数组替换原数组
    clear_array = ROOT.ROC_party_tokens
    for_each_loop = {
        array = temp_ROC_party_tokens_reordered
        value = temp_roc_token
        add_to_array = { ROC_party_tokens = var:temp_roc_token }
    }
    
    # 清理临时数组
    clear_array = temp_ROC_party_tokens_reordered
}

# 用法：
#   set_temp_variable = { parliament_adjust_party = token:parliament_gui_KMT }  # 目标政党
#   set_temp_variable = { parliament_adjust_value = 10 }  # 正数增加、负数减少
#   parliament_adjust_seats = yes
parliament_adjust_seats = {
    set_temp_variable = { target_party = parliament_adjust_party }
    set_temp_variable = { adjust_value = parliament_adjust_value }
    
    if = {
        limit = {
            check_variable = { adjust_value < 0 }
        }
        parliament_decrease_seats = yes
    }
    else = {
        parliament_increase_seats = yes
    }
    
    # 清理临时变量
    clear_temp_variable = target_party
    clear_temp_variable = adjust_value
}

# 增加政党议席（随机从其他政党抽取）
parliament_increase_seats = {
    set_temp_variable = { seats_to_add = parliament_adjust_value }
    
    while = {
        limit = {
            check_variable = { seats_to_add > 0 }
        }
        
        random_list = {
            parliament_select_random_donor = {
                weight = 1
                parliament_select_donor_party = yes
            }
        }
        
        # 从捐献方减少1席，给目标政党增加1席
        if = {
            limit = {
                set_temp_variable = { donor_party = var:parliament_selected_donor }
                check_variable = { 
                    donor_party > 0  # 确保找到了捐献方
                }
            }
            meta_effect = {
                text = {
                    [donor_token] = {
                        add_to_variable = {
                            parliament_gui_seats = -1
                        }
                    }
                    [target_token] = {
                        add_to_variable = {
                            parliament_gui_seats = 1
                        }
                    }
                }
                donor_token = "[?parliament_selected_donor.GetTokenKey]"
                target_token = "[?parliament_adjust_party.GetTokenKey]"
            }
        }
        
        add_to_temp_variable = { seats_to_add = -1 }
    }
}

# 减少政党议席（随机分配给其他政党）
parliament_decrease_seats = {
    # 将负数转正
    set_temp_variable = { seats_to_remove = parliament_adjust_value }
    if = {
        limit = {
            check_variable = { seats_to_remove < 0 }
        }
        multiply_temp_variable = { seats_to_remove = -1 }
    }
    
    # 循环分配被减少的议席
    while = {
        limit = {
            check_variable = { seats_to_remove > 0 }
        }
        
        # 随机选择一个接收方政党
        random_list = {
            parliament_select_random_recipient = {
                weight = 1
                parliament_select_recipient_party = yes
            }
        }
        
        # 从目标政党减少1席，给接收方增加1席
        if = {
            limit = {
                set_temp_variable = { recipient_party = parliament_selected_recipient }
                check_variable = {
                    recipient_party > 0
                }
            }
            meta_effect = {
                text = {
                    [target_token] = {
                        add_to_variable = {
                            parliament_gui_seats = -1
                        }
                    }
                    [recipient_token] = {
                        add_to_variable = {
                            parliament_gui_seats = 1
                        }
                    }
                }
                target_token = "[?parliament_adjust_party.GetTokenKey]"
                recipient_token = "[?parliament_selected_recipient.GetTokenKey]"
            }
        }
        
        add_to_temp_variable = { seats_to_remove = -1 }
    }
}

# 从活跃政党中随机选择一个捐献方（不能是目标政党）
parliament_select_donor_party = {
    set_temp_variable = { parliament_selected_donor = 0 }
    set_temp_variable = { selection_attempts = 0 }
    
    while = {
        limit = {
            check_variable = { parliament_selected_donor = 0 }
            check_variable = { selection_attempts < 20 }  # 防止无限循环
        }
        
        # 随机选择数组中的一个政党
        random_list = {
            parliament_pick_random = {
                weight = 1
                for_each_loop = {
                    array = ROC_party_tokens
                    value = roc_random_token
                    meta_effect = {
                        text = {
                            set_temp_variable = { temp_selected_token = [tokenbase] }
                        }
                        tokenbase = "token:[?roc_random_token.GetTokenKey]"
                    }
                }
            }
        }
        
        # 检查选中的党派是否有议席且不是目标政党
        meta_trigger = {
            text = {
                [tokenbase] = {
                    OR = {
                        check_variable = {
                            parliament_gui_status = 2  # 联合党
                        }
                        check_variable = {
                            parliament_gui_status = 3  # 在野党
                        }
                    }
                    check_variable = {
                        parliament_gui_seats > 0
                    }
                    NOT = {
                        check_variable = {
                            parliament_gui_party = parliament_adjust_party
                        }
                    }
                }
            }
            tokenbase = "[?temp_selected_token.GetTokenKey]"
        }
        meta_effect = {
            text = {
                set_temp_variable = { parliament_selected_donor = [tokenbase] }
            }
            tokenbase = "token:[?temp_selected_token.GetTokenKey]"
        }
        
        add_to_temp_variable = { selection_attempts = 1 }
    }
}

# 随机选择一个接收方政党（不能是目标政党）
parliament_select_recipient_party = {
    set_temp_variable = { parliament_selected_recipient = 0 }
    set_temp_variable = { selection_attempts = 0 }
    
    while = {
        limit = {
            check_variable = { parliament_selected_recipient = 0 }
            check_variable = { selection_attempts < 20 }
        }
        
        # 随机选择数组中的一个政党
        random_list = {
            parliament_pick_random_recipient = {
                weight = 1
                for_each_loop = {
                    array = ROC_party_tokens
                    value = roc_random_token
                    meta_effect = {
                        text = {
                            set_temp_variable = { temp_selected_recipient = [tokenbase] }
                        }
                        tokenbase = "token:[?roc_random_token.GetTokenKey]"
                    }
                }
            }
        }
        
        # 检查选中的党派不是目标政党
        meta_trigger = {
            text = {
                [tokenbase] = {
                    NOT = {
                        check_variable = {
                            parliament_gui_party = var:parliament_adjust_party
                        }
                    }
                }
            }
            tokenbase = "[?temp_selected_recipient.GetTokenKey]"
        }
        meta_effect = {
            text = {
                set_temp_variable = { parliament_selected_recipient = [tokenbase] }
            }
            tokenbase = "token:[?temp_selected_recipient.GetTokenKey]"
        }
        
        add_to_temp_variable = { selection_attempts = 1 }
    }
}

ROC_parliament_update = {
	set_temp_variable = { tempc = 0 }
	for_each_loop = {
		array = ROC_party_tokens
		value = roc_cur_token
		ROOT = {
			set_variable = { tempc = roc_cur_token }
			set_variable = { roc_seats = 0 }
			set_variable = { roc_frame = 0 }
		}
		meta_effect = {
			text = {
				every_character = {
					limit = { is_character = [tokenbase] }
					set_temp_variable = { roc_frame = THIS.parliament_gui_frame }
					set_temp_variable = { roc_seats = THIS.parliament_gui_seats }
				}
			}
			tokenbase = "[?ROOT.tempc.GetTokenKey]"
		}
		log = "[tokenbase]: seats = [?roc_seats], frame = [?roc_frame]"
		meta_effect = {
			text = {
				set_temp_variable = {
					t[tier] = [party]
				}
				log = "[tier] [party]"
			}
			tier = "[?roc_frame]"
			party = "[?roc_seats]"
		}
		ROOT = {
			clear_variable = ROOT.tempc
			clear_variable = roc_seats
			clear_variable = roc_frame
		}
	}
	log = "[?t1] [?t2] [?t3] [?t4] [?t5] [?t6] [?t7] [?t8] [?t9] [?t10]"

	add_to_temp_variable = { t2 = t1 }
	add_to_temp_variable = { t3 = t2 }
	add_to_temp_variable = { t4 = t3 }
	add_to_temp_variable = { t5 = t4 }
	add_to_temp_variable = { t6 = t5 }
	add_to_temp_variable = { t7 = t6 }
	add_to_temp_variable = { t8 = t7 }
	add_to_temp_variable = { t9 = t8 }
	set_temp_variable = { t10 = 800 }
	log = "[?t1] [?t2] [?t3] [?t4] [?t5] [?t6] [?t7] [?t8] [?t9] [?t10]"
	
	clear_array = ROC_seat_frame

	set_temp_variable = { ROC_resize_array_times = 0 }
	log = "[?ROC_resize_array_times] [?ROC_parties_max]"
	while_loop_effect = {
		limit = {
			check_variable = {
				ROC_resize_array_times < ROC_parties_max
			}
		}
		add_to_temp_variable = { ROC_resize_array_times = 1 }
		meta_effect = {
			text = {
				resize_array = {
					array = ROC_seat_frame
					value = [time]
					size = t[time]
				}
			}
			time = "[?ROC_resize_array_times]"
		}
		log = "[?ROC_resize_array_times]"
	}
}

ROC_init_coalition_construct = {
	set_country_flag = open_roc_coalition_gui
	set_temp_variable = { tempc = 0 }
	for_each_loop = {
		array = ROC_party_tokens
		value = roc_cur_token
		ROOT = {
			set_variable = { tempc = roc_cur_token }
		}
		meta_effect = {
			text = {
				every_character = {
					limit = { is_character = [tokenbase] }
					if = {
						limit = {
							check_variable = {
								THIS.parliament_gui_status = 1
							}
						}
						set_variable = { parliament_gui_constructalig = 1 }
						set_variable = { parliament_gui_interacted = 1 }
					}
					else_if = {
						limit = {
							check_variable = {
								THIS.parliament_gui_status = 2
							}
						}
						set_variable = { parliament_gui_constructalig = 2 }
						set_variable = { parliament_gui_interacted = 1 }
					}
				}
			}
			tokenbase = "[?ROOT.tempc.GetTokenKey]"
		}
		ROOT = {
			set_variable = { tempc = 0 }
		}
	}
}

ROC_coalition_complete = {
	set_temp_variable = { tempc = 0 }
	for_each_loop = {
		array = ROC_party_tokens
		value = roc_cur_token
		ROOT = {
			set_variable = { tempc = roc_cur_token }
		}
		meta_effect = {
			text = {
				[tokenbase] = {
					if = {
						limit = {
							check_variable = {
								THIS.parliament_gui_constructalig = 1
							}
						}
						set_variable = { parliament_gui_status = 1 }
						clear_variable = parliament_gui_interacted
						clear_variable = parliament_gui_constructalig
					}
					else_if = {
						limit = {
							check_variable = {
								THIS.parliament_gui_constructalig = 2
							}
						}
						set_variable = { parliament_gui_status = 2 }
						clear_variable = parliament_gui_interacted
						clear_variable = parliament_gui_constructalig
						set_temp_variable = { coalition_partner_var = var:parliament_gui_ideology }
						add_to_coalition = yes
						log = "1145141"
					}
					else_if = {
						limit = {
							check_variable = {
								THIS.parliament_gui_constructalig = 3
							}
						}
						set_variable = { parliament_gui_status = 3 }
						clear_variable = parliament_gui_interacted
						clear_variable = parliament_gui_constructalig
						set_temp_variable = { coalition_partner_var = var:parliament_gui_ideology }
						remove_from_coalition = yes
					}
				}
			}
			tokenbase = "[?ROOT.tempc.GetTokenKey]"
		}
		ROOT = {
			set_variable = { tempc = 0 }
		}
	}
	meta_effect = {
		text = {
			add_political_power = [pp]
		}
		pp = "-[GetROCCoalitionAllPPCost]"
	}
	# ROC_parliament_reorder_parties = yes
	ROC_parliament_update = yes
}
