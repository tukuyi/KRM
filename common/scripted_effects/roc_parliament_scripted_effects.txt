ROC_parliament_parties_num_check = {
	set_temp_variable = { temp_parties_num = 0 }
	for_each_loop = {
		array = ROC_party_tokens
		value = roc_cur_token

	set_variable = { ROC_parties_max = var:temp_parties_num }
}


ROC_parliament_reorder_parties = {
    clear_array = ROOT.ROC_party_tokens
    
    set_temp_variable = { ruling_party_found = 0 }
	every_character = {
		limit = {
			check_variable = {
				THIS.parliament_gui_status = 1
			}
		}
		add_to_array = { ROOT.ROC_party_tokens = var:parliament_gui_party }
		set_temp_variable = { ruling_party_found = 1 }
	}
	every_character = {
		limit = {
			check_variable = {
				THIS.parliament_gui_status = 2
			}
		}
		add_to_array = { ROOT.ROC_party_tokens = var:parliament_gui_party }
	}
    
	every_character = {
		limit = {
			check_variable = {
				THIS.parliament_gui_status = 3
			}
			NOT = {
				is_character = parliament_gui_IND
			}
		}
		add_to_array = { ROOT.ROC_party_tokens = var:parliament_gui_party }
	}

	every_character = {
		limit = {
			is_character = parliament_gui_IND
		}
		add_to_array = { ROOT.ROC_party_tokens = var:parliament_gui_party }
	}
}


DEBUG_ROC_parliament_remove_YCPCPAC_seats = {
  set_temp_variable = { parliament_adjust_party = token:parliament_gui_YCPCPAC }
  set_temp_variable = { parliament_adjust_value = -30 }
  parliament_adjust_seats = yes
}

# 用法：
#   set_temp_variable = { parliament_adjust_party = token:parliament_gui_KMT }
#   set_temp_variable = { parliament_adjust_value = 10 }
#   parliament_adjust_seats = yes

parliament_adjust_seats = {
    if = {
        limit = {
            check_variable = { parliament_adjust_value < 0 }
        }
        parliament_decrease_seats = yes
    }
    else = {
        parliament_increase_seats = yes
    }
	ROC_parliament_update = yes
}

# 增加政党议席（随机从其他政党抽取）
parliament_increase_seats = {
    set_temp_variable = { temp_seats_remaining = var:parliament_adjust_value }
    
    parliament_transfer_seat_to_target = yes
}

# 减少政党议席（随机分配给其他政党）
parliament_decrease_seats = {
    set_temp_variable = { temp_seats_to_distribute = var:parliament_adjust_value }
    if = {
        limit = {
            check_variable = { temp_seats_to_distribute < 0 }
        }
        multiply_temp_variable = { temp_seats_to_distribute = -1 }
    }
    
    parliament_transfer_seat_from_target = yes
}

# 从其他政党转移1席给目标政党（增加）
parliament_transfer_seat_to_target = {
    if = {
        limit = {
            check_variable = { temp_seats_remaining > 0 }
        }
		set_temp_variable = { temp_select_party = 0 }
		while_loop_effect = {
			limit = {
				check_variable = { temp_seats_remaining > 0 }
			}
			set_temp_variable = { temp_parties_num = var:ROC_parties_max }
			add_to_temp_variable = { temp_parties_num = 1 }
			set_temp_variable_to_random = {
				var = temp_select_party
				min = 1
				max = var:temp_parties_num
				integer = yes
			}
			add_to_temp_variable = { temp_select_party = -1 }
			meta_effect = {
				text = {
					set_temp_variable = { temp_current_party = var:ROC_party_tokens^[tokenbase] }
				}
				tokenbase = "[?temp_select_party]"
			}
			meta_effect = {
				text = {
					[tokenbase] = {
						if = {
							limit = {
								check_variable = { THIS.parliament_gui_seats > 0 }
								NOT = { check_variable = { THIS.parliament_gui_party = [target_party_token] } }
							}
							add_to_variable = { parliament_gui_seats = -1 }
							PREV = {
								[target_party] = {
									add_to_variable = { parliament_gui_seats = 1 }
								}
								add_to_temp_variable = { temp_seats_remaining = -1 }
							}
						}
					}
				}
				tokenbase = "[?temp_current_party.GetTokenKey]"
				target_party = "[?parliament_adjust_party.GetTokenKey]"
				target_party_token = "[?parliament_adjust_party]"
			}
		}
    }
}

# 从目标政党转移1席给其他政党（减少）
parliament_transfer_seat_from_target = {
    if = {
        limit = {
            check_variable = { temp_seats_to_distribute > 0 }
        }
		set_temp_variable = { temp_select_party = 0 }
		while_loop_effect = {
			limit = {
				check_variable = { temp_seats_to_distribute > 0 }
				meta_trigger = {
					text = {
						[tokenbase] = {
							check_variable = { THIS.parliament_gui_seats > 0 }
						}
					}
					tokenbase = "[?parliament_adjust_party.GetTokenKey]"
				}
			}
			set_temp_variable = { temp_parties_num = var:ROC_parties_max }
			add_to_temp_variable = { temp_parties_num = 1 }
			set_temp_variable_to_random = {
				var = temp_select_party
				min = 1
				max = var:temp_parties_num
				integer = yes
			}
			add_to_temp_variable = { temp_select_party = -1 }
			meta_effect = {
				text = {
					set_temp_variable = { temp_current_party = var:ROC_party_tokens^[tokenbase] }
				}
				tokenbase = "[?temp_select_party]"
			}
			meta_effect = {
				text = {
					[tokenbase] = {
						if = {
							limit = {
								NOT = { check_variable = { THIS.parliament_gui_party = [target_party_token] } }
							}
							add_to_variable = { parliament_gui_seats = 1 }
							PREV = {
								[target_party] = {
									add_to_variable = { parliament_gui_seats = -1 }
								}
								add_to_temp_variable = { temp_seats_to_distribute = -1 }
							}
						}
					}
				}
				tokenbase = "[?temp_current_party.GetTokenKey]"
				target_party = "[?parliament_adjust_party.GetTokenKey]"
				target_party_token = "[?parliament_adjust_party]"
			}
		}
    }
}

ROC_parliament_update = {
	set_temp_variable = { tempc = 0 }
    set_temp_variable = { ROC_loop_index = 0 }          # ← 新增：手动计数器

    for_each_loop = {
        array = ROC_party_tokens
        value = roc_cur_token

        add_to_temp_variable = { ROC_loop_index = 1 }   # ← 新增：每轮递增

        ROOT = {
            set_variable = { tempc = roc_cur_token }
            set_variable = { roc_seats = 0 }
            set_variable = { roc_frame = 0 }
        }
        meta_effect = {
            text = {
                every_character = {
                    limit = { is_character = [tokenbase] }
                    set_temp_variable = { roc_seats = THIS.parliament_gui_seats }
					set_temp_variable = { roc_frame = THIS.parliament_gui_frame }  # ← 恢复
                }
            }
            tokenbase = "[?ROOT.tempc.GetTokenKey]"
        }
        meta_effect = {
            text = {
                set_temp_variable = {
                    t[idx] = [seats]          # ← 下标改为循环计数器
                }
				set_temp_variable = { f[idx] = [frame_val] }   # ← 新增
            }
            idx   = "[?ROC_loop_index]"       # ← 原来是 roc_frame
            seats = "[?roc_seats]"
  			frame_val = "[?roc_frame]"                         # ← 新增	
        }
		if = {
			limit = {
				check_variable = { ROC_loop_index = var:ROC_parties_max }
			}
			meta_effect = {
				text = {
					set_temp_variable = { t[idx] = var:ROC_seats_max }
				}
				idx = "[?ROC_loop_index]"
			}
		}
		else_if = {
			limit = {
				check_variable = { ROC_loop_index > 1 }
			}
			set_temp_variable = { temp_index_calculate = var:ROC_loop_index }
			add_to_temp_variable = { temp_index_calculate = -1 }
			meta_effect = {
				text = {
					add_to_temp_variable = { t[idx] = t[calidx] }          # ← 下标改为循环计数器
				}
				idx = "[?ROC_loop_index]"
				calidx = "[?temp_index_calculate]"
			}
		}
    }
	ROOT = {
		clear_variable = ROOT.tempc
		clear_variable = roc_seats
		# clear roc_frame 也删去
	}
	
	clear_array = ROC_seat_frame

	set_temp_variable = { ROC_resize_array_times = 0 }
	while_loop_effect = {
		limit = {
			check_variable = {
				ROC_resize_array_times < ROC_parties_max
			}
		}
		add_to_temp_variable = { ROC_resize_array_times = 1 }
		meta_effect = {
			text = {
				resize_array = {
					array = ROC_seat_frame
					value = f[time]
					size = t[time]
				}
			}
			time = "[?ROC_resize_array_times]"
		}
	}
	ROC_parliament_parties_num_check = yes
}

##############################################################
# 国民大会 — 选举结算
#
# 调用前在 ROC_register_parties 中为各党设置：
#   clear_array = parliament_gui_ideologies
#   add_to_array = { parliament_gui_ideologies = token:authoritarian_democrat }
#   add_to_array = { parliament_gui_ideologies = token:social_conservative }   # 可多个
#
# 无党籍支持率自动计算，无需手动设置。
#
# 可选参数（国家变量，调用前设置）：
#   ROC_election_noise_pct   随机波动幅度，默认 0.1（即 ±10%）
#   ROC_election_min_seats   最低保障席位，默认 3
#   ROC_election_ideo_penalty 多意识形态惩罚系数，默认 0.02
##############################################################
 
ROC_election_calculate = {
 
    # ── 默认参数 ──────────────────────────────────────────
    if = {
        limit = { NOT = { check_variable = { ROC_election_noise_pct > 0 } } }
        set_temp_variable = { ROC_election_noise_pct = 0.08 }
    }
    else = {
        set_temp_variable = { ROC_election_noise_pct = var:ROC_election_noise_pct }
    }
    if = {
        limit = { NOT = { check_variable = { ROC_election_min_seats > 0 } } }
        set_temp_variable = { ROC_election_min_seats = 3 }
    }
    else = {
        set_temp_variable = { ROC_election_min_seats = var:ROC_election_min_seats }
    }
    if = {
        limit = { NOT = { check_variable = { ROC_election_ideo_penalty > 0 } } }
        set_temp_variable = { ROC_election_ideo_penalty = 0.02 }
    }
    else = {
        set_temp_variable = { ROC_election_ideo_penalty = var:ROC_election_ideo_penalty }
    }
    set_temp_variable = { elec_seats_max = var:ROC_seats_max }
 
    # ── 预缓存所有意识形态人气为同名 temp 变量 ───────────
    # meta 展开时可用 party_popularity_[ideo] 直接引用
    set_temp_variable = { party_popularity_totalist              = party_popularity@totalist              }
    set_temp_variable = { party_popularity_syndicalist           = party_popularity@syndicalist           }
    set_temp_variable = { party_popularity_radical_socialist     = party_popularity@radical_socialist     }
    set_temp_variable = { party_popularity_social_democrat       = party_popularity@social_democrat       }
    set_temp_variable = { party_popularity_social_liberal        = party_popularity@social_liberal        }
    set_temp_variable = { party_popularity_market_liberal        = party_popularity@market_liberal        }
    set_temp_variable = { party_popularity_social_conservative   = party_popularity@social_conservative   }
    set_temp_variable = { party_popularity_authoritarian_democrat = party_popularity@authoritarian_democrat }
    set_temp_variable = { party_popularity_paternal_autocrat     = party_popularity@paternal_autocrat     }
    set_temp_variable = { party_popularity_national_populist     = party_popularity@national_populist     }
 
    # ══════════════════════════════════════════════════════
    # 第零步：自动计算无党籍支持率
    #
    # extremism = avg(totalist, national_populist,
    #                 syndicalist, radical_socialist)
    # frag      = 1 - 各党原始权重中的最大值
    # result    = clamp((extremism + frag) / 2, 0, 0.33)
    # ══════════════════════════════════════════════════════
 
    # 极端化分量
    set_temp_variable = { ROC_ind_extreme_sum = 0 }
    add_to_temp_variable = { ROC_ind_extreme_sum = party_popularity@totalist          }
    add_to_temp_variable = { ROC_ind_extreme_sum = party_popularity@national_populist }
    add_to_temp_variable = { ROC_ind_extreme_sum = party_popularity@syndicalist       }
    add_to_temp_variable = { ROC_ind_extreme_sum = party_popularity@radical_socialist }
    divide_temp_variable  = { ROC_ind_extreme_sum = 4 }
 
    # 碎片化分量：遍历各党，用多意识形态求和计算原始权重，找最大值
    set_temp_variable = { ROC_ind_max_support = 0 }
    for_each_loop = {
        array = ROC_party_tokens
        value = roc_cur_token
        meta_effect = {
            text = {
                every_character = {
                    limit = {
                        is_character = [party]
                        NOT = { is_character = parliament_gui_IND }
                    }
                    # 遍历该党意识形态数组求和
                    set_temp_variable = { temp_raw_w = 0 }
                    set_temp_variable = { temp_ideo_n = 0 }
                    for_each_loop = {
                        array = parliament_gui_ideologies
                        value = cur_ideo
						ROC_cal_cur_ideo_effect = yes
                        add_to_temp_variable = { temp_ideo_n = 1 }
                    }
                    # 多意识形态惩罚
                    add_to_temp_variable = { temp_ideo_n = -1 }
                    multiply_temp_variable = { temp_ideo_n = var:ROC_election_ideo_penalty }
                    subtract_from_temp_variable = { temp_raw_w = var:temp_ideo_n }
					# log = "[party]"
					# log = "[?temp_raw_w]"
                    # 更新最大值
                    if = {
                        limit = { check_variable = { temp_raw_w > var:ROC_ind_max_support } }
                        set_temp_variable = { ROC_ind_max_support = var:temp_raw_w }
						# log = "[ROC_ind_max_support] from [party]"
                    }
                }
            }
            party = "[?roc_cur_token.GetTokenKey]"
            ideo = "[?cur_ideo.GetTokenKey]"
        }
    }
 
    # 合并 + 钳位
    set_variable = { ROC_support_independent = 1 }
    subtract_from_variable = { ROC_support_independent = var:ROC_ind_max_support }
    add_to_variable = { ROC_support_independent = var:ROC_ind_extreme_sum }
    divide_variable  = { ROC_support_independent = 4 }
    if = {
        limit = { check_variable = { ROC_support_independent > 0.33 } }
        set_variable = { ROC_support_independent = 0.33 }
    }
    if = {
        limit = { check_variable = { ROC_support_independent < 0 } }
        set_variable = { ROC_support_independent = 0 }
    }
	# log = "无党籍支持率： [?ROC_support_independent]"
 
    # ══════════════════════════════════════════════════════
    # 第一遍：计算各党加噪权重，存入 character.elec_w
    #         累计总权重，追踪最大权重党
    # ══════════════════════════════════════════════════════
    set_temp_variable = { ROC_elec_total_weight = 0 }
    set_temp_variable = { ROC_elec_max_w        = 0 }
 
    for_each_loop = {
        array = ROC_party_tokens
        value = roc_cur_token
        meta_effect = {
            text = {
                every_character = {
                    limit = { is_character = [party] }
					set_temp_variable = { parliament_gui_elec_w = 0 }
 
                    # 无党籍：直接用预算好的 ROC_support_independent
                    if = {
                        limit = { is_character = parliament_gui_IND }
                        set_temp_variable = { elec_base = var:ROOT.ROC_support_independent }
                    }
                    else = {
                        # 遍历意识形态数组求和 + 惩罚
                        set_temp_variable = { elec_base   = 0 }
                        set_temp_variable = { elec_ideo_n = 0 }
                        for_each_loop = {
                            array = parliament_gui_ideologies
                            value = cur_ideo
                            ROC_cal_elec_base_effect = yes
                            add_to_temp_variable = { elec_ideo_n = 1 }
                        }
                        add_to_temp_variable = { elec_ideo_n = -1 }
                        multiply_temp_variable = { elec_ideo_n = var:ROC_election_ideo_penalty }
                        subtract_from_temp_variable = { elec_base = var:elec_ideo_n }
                    }
					# 噪声：base × rand([-pct, +pct])
					set_temp_variable = { elec_noise_2x = var:ROC_election_noise_pct }
					multiply_temp_variable = { elec_noise_2x = 2 }
					set_temp_variable_to_random = {
						var = elec_roll
						min = 0
						max = var:elec_noise_2x
					}
					subtract_from_temp_variable = { elec_roll = var:ROC_election_noise_pct }
					set_temp_variable = { elec_delta = var:elec_base }
					multiply_temp_variable = { elec_delta = var:elec_roll }
			
					set_temp_variable = { elec_w = var:elec_base }
					add_to_temp_variable = { elec_w = var:elec_delta }
			
					add_to_temp_variable = { parliament_gui_elec_w = var:elec_w }
					add_to_temp_variable = { ROC_elec_total_weight = var:parliament_gui_elec_w }
					# log = "[?elec_s] [?THIS.elec_w] [?ROOT.elec_w]"
					add_to_temp_variable = { ROC_elec_total_weight = PREV.elec_w }
					if = {
						limit = { check_variable = { PREV.elec_w > THIS.ROC_elec_max_w } }
						set_temp_variable = { ROC_elec_max_w = PREV.elec_w }
						set_variable = { ROC_elec_max_party = var:roc_cur_token }
						# log = "114514 [?ROC_elec_max_party]"
					}
                }
            }
            party = "[?roc_cur_token.GetTokenKey]"
        }
		# 存到 character 身上供第二遍读取
        meta_effect = {
            text = {
                every_character = {
                    limit = { is_character = [party] }
					set_variable = { elec_s = var:parliament_gui_elec_w }
                }
            }
            party = "[?roc_cur_token.GetTokenKey]"
        }
    }
 
    # ══════════════════════════════════════════════════════
    # 第二遍：按权重比例换算席位，写入 parliament_gui_seats
    # ══════════════════════════════════════════════════════
    set_temp_variable = { elec_total_w    = var:ROC_elec_total_weight }
	log = "Total election weight: [?elec_seats_max]"
    set_temp_variable = { ROC_elec_seats_sum = 0 }
 
    for_each_loop = {
        array = ROC_party_tokens
        value = roc_cur_token
        meta_effect = {
            text = {
                every_character = {
                    limit = { is_character = [party] }
                    multiply_variable = { elec_s = var:elec_seats_max }
                    divide_variable   = { elec_s = elec_total_w }
                    round_variable    = elec_s
                    if = {
                        limit = { check_variable = { elec_s < var:elec_min_seats } }
                        set_variable = { elec_s = var:elec_min_seats }
                    }
                    set_variable = { parliament_gui_seats = var:elec_s }
                    ROOT = { add_to_temp_variable = { ROC_elec_seats_sum = PREV.parliament_gui_seats } }
                    clear_variable = elec_s
                    clear_variable = elec_w
                }
            }
            party = "[?roc_cur_token.GetTokenKey]"
        }
    }
 
    # ── 余数修正 ──────────────────────────────────────────
    set_variable = { ROC_elec_remainder = var:ROC_seats_max }
    subtract_from_variable = { ROC_elec_remainder = var:ROC_elec_seats_sum }
    meta_effect = {
        text = {
            every_character = {
                limit = { is_character = [party] }
                set_temp_variable = { temp_remainder = ROOT.ROC_elec_remainder }
                add_to_variable = { parliament_gui_seats = temp_remainder }
            }
        }
        party = "[?ROC_elec_max_party.GetTokenKey]"
    }
    set_variable = { ROC_elec_remainder = 0 }

	set_variable = { ROC_support_independent = 0 }
 
    ROC_parliament_reorder_parties = yes
    ROC_parliament_update = yes
}
