HNN_no_cosmetic_flag_check = {
	NOT = {
		has_cosmetic_tag = HNN_SCPC
		has_cosmetic_tag = united_china
	}
}
uses_governorate_autonomy_state = {
	overlord = { is_chinese_tag = yes }
	is_chinese_tag = yes
}

SZC_KMT_is_socialist_or_allied_trigger = {
	OR = {
		AND = {
			is_ally_with = SZC
		}
		AND = {
			SZC = {
				has_socialist_government = yes
			}
			has_socialist_government = yes
		}
	}
}
SZC_KMT_FROM_has_compatible_government_trigger = {
	#Remember this is scoped to FROM
	is_monarchy = no
	OR = {
		#South East Asia
		AND = {
			tag = BRM
			NOT = { ROOT = { is_ally_with = SIA } } #Burma won't ally with KMT if they are allied with Siam
			OR = {
				has_elected_government = yes
				has_government = paternal_autocrat
			}
		}
		AND = {
			tag = PHI
		}
		AND = {
			tag = INS
			has_country_leader = { character = INS_sukarno }
		}
		AND = {
			OR = {
				tag = VIN
				tag = INC
			}
			INC_has_VNQDD_government_trigger = yes
		}

		#Middle East
		AND = {
			tag = TUR
			OR = {
				has_country_leader = { character = TUR_ismet_inonu }
				has_country_leader = { character = TUR_mehmet_recep_peker }
			}
		}
	}
}
